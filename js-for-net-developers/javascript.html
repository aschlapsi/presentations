Objekte in JavaScript...

  • Alles außer primitive Datentypen in JavaScript ist ein Objekt.
  • Sie beinhalten Eigenschaften (Key/Value-Pairs).
  • Sie sind von außen veränderbar.
  • Sie sind vergleichbar mit System.Collections.HashTable.

that statt this

Self-invoking Constructor

        function Animal(name) {
          if (!(this instanceof Animal)) {
            return new Animal(name);
          }

          this.name = name;
        }

        var pet = Animal("Murli");
        console.log(pet.name);
        console.log(name);

Vererbung mit Prototypes

        function Pet(name, species) {
          this.name = name;
          this.species = species;
        }
        Pet.prototype.greet = function() {
          console.log("I'm " + this.name + ".");
          console.log("I'm a " + this.species + ".");
        }
        function Dog(name) {
          Pet.call(this, name, "Dog");
        }
        Dog.prototype = new Pet();
        Dog.prototype.constructor = Dog;
        var dog = new Dog("Waldi");
        dog.greet();
